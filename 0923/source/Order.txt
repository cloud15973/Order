package com;

public class Order {
	private String name;
	private int babyformula;
	private int diaper;
	private int sum;

	public Order(String name, int babyformula, int diaper) {
		super();
		this.name = name;
		this.babyformula = babyformula;
		this.diaper = diaper;
		this.sum = babyformula * 829 + diaper * 658;
	}

	public String getName() {
		return name;
	}

	public void setName(String name) {
		this.name = name;
	}

	public int getBabyformula() {
		return babyformula;
	}

	public void setBabyformula(int babyformula) {
		this.babyformula = babyformula;
	}

	public int getDiaper() {
		return diaper;
	}

	public void setDiaper(int diaper) {
		this.diaper = diaper;
	}

	public int getSum() {
		return sum;
	}

}
